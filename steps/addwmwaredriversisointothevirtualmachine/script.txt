
virtualMachine="{virtualMachine.fqn}"
storagePool="{storagePool}"

pwsh <<'EOF'
$ErrorActionPreference = "Stop"
Import-Module VMware.VimAutomation.Core

Connect-VIServer {vcenterHost.ip} `
    -User {vcenterUser.user} `
    -Password {vcenterUser.password}

$vm = Get-VM -Name "$virtualMachine"
$iso = "[$storagePool] kickstart_isos/$virtualMachine-Drivers.iso"

New-CDDrive -VM $vm -IsoPath $iso -StartConnected

EOF
