
virtualMachine="{virtualMachine.fqn}"
storagePool="{storagePool}"

# Create the new VM
pwsh <<'EOF'
$ErrorActionPreference = "Stop"
Import-Module VMware.VimAutomation.Core

Connect-VIServer {vcenterHost.ip} `
    -User {vcenterUser.user} `
    -Password {vcenterUser.password}

$cd = Get-VM -Name "$virtualMachine" | Get-CDDrive
$iso = "[$storagePool] kickstart_isos/kickstart_$virtualMachine.iso"

Set-CDDrive -CD $cd `
    -IsoPath $iso `
    -StartConnected $true `
    -Confirm:$false

EOF
